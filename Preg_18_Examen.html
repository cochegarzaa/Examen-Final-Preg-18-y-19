# Create DataFrame
df = pd.DataFrame(data)

# Step 1: Data Preparation - Compute log DPI
df['log_DPI'] = np.log(df['PDI'])

# Step 2: Visual Inspection - Plot the log DPI series
plt.figure(figsize=(12, 6))
plt.plot(df['YEAR'], df['log_DPI'], label='Log DPI')
plt.title('Log of Personal Disposable Income (1970.1 - 1991.4)')
plt.xlabel('Year')
plt.ylabel('Log DPI')
plt.xticks(df['YEAR'][::4], rotation=45)  # Show every 4th year for readability
plt.legend()
plt.show()

# Step 3: Stationarity Check - ADF Test
def adf_test(series, name):
    result = adfuller(series)
    print(f"\nADF Test for {name}:")
    print(f'ADF Statistic: {result[0]:.4f}')
    print(f'p-value: {result[1]:.4f}')
    print(f'Critical Values: {result[4]}')
    if result[1] <= 0.05:
        print("Result: Stationary (reject null hypothesis)")
    else:
        print("Result: Non-stationary (fail to reject null hypothesis)")

adf_test(df['log_DPI'], "Log DPI")

# First difference if non-stationary
df['log_DPI_diff'] = df['log_DPI'].diff().dropna()
adf_test(df['log_DPI_diff'].dropna(), "First Differenced Log DPI")

# Step 4: Identify ARIMA Model - Plot ACF and PACF
plt.figure(figsize=(12, 6))
plt.subplot(211)
plot_acf(df['log_DPI_diff'].dropna(), ax=plt.gca(), lags=20)
plt.title('ACF of First Differenced Log DPI')
plt.subplot(212)
plot_pacf(df['log_DPI_diff'].dropna(), ax=plt.gca(), lags=20)
plt.title('PACF of First Differenced Log DPI')
plt.tight_layout()
plt.show()

# Step 5: Estimate the ARIMA Model (Tentative ARIMA(1,1,1) based on typical patterns)
model = ARIMA(df['log_DPI'], order=(1, 1, 1))
results = model.fit()
print("\nARIMA(1,1,1) Model Summary:")
print(results.summary())

# Step 6: Model Diagnostics - Residual Analysis
residuals = results.resid
