# Funcin para encontrar el mejor modelo ARMA (usando ARIMA con d=0)
def find_best_arma(series, name, max_p=3, max_q=3):
    best_aic = float('inf')
    best_order = None
    for p in range(max_p + 1):
        for q in range(max_q + 1):
            try:
                model = ARIMA(series, order=(p, 0, q))
                results = model.fit()
                if results.aic < best_aic:
                    best_aic = results.aic
                    best_order = (p, 0, q)
            except:
                continue
    print(f"\nMejor modelo ARMA para {name}:")
    print(f"Orden: {best_order}")
    print(f"AIC: {best_aic:.2f}")
    best_model = ARIMA(series, order=best_order).fit()
    return best_model

# Ajustar modelos ARMA

# Funcin para encontrar el mejor modelo ARIMA
def find_best_arima(series, name, max_p=3, max_d=2, max_q=3):
    best_aic = float('inf')
    best_order = None
    for p in range(max_p + 1):
        for d in range(max_d + 1):
            for q in range(max_q + 1):
                try:
                    model = ARIMA(series, order=(p, d, q))
                    results = model.fit()
                    if results.aic < best_aic:
                        best_aic = results.aic
                        best_order = (p, d, q)
                    print(f"Probando orden {p, d, q} - AIC: {results.aic:.2f}")
                except Exception as e:
                    print(f"Error con orden {p, d, q}: {e}")
                    continue
    if best_order is None:
        raise ValueError(f"No se pudo ajustar un modelo ARIMA para {name}")
    print(f"\nMejor modelo ARIMA para {name}:")
    print(f"Orden: {best_order}")
    print(f"AIC: {best_aic:.2f}")
    return best_order
